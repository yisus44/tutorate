
DROP DATABASE tutorate;
CREATE DATABASE tutorate;
USE tutorate;

CREATE TABLE Usuarios
(
UsuarioID INT NOT NULL AUTO_INCREMENT,
Nombre VARCHAR(50),
Email VARCHAR(30),
Edad INT,
Contrasena VARCHAR(50),
PRIMARY KEY(UsuarioID)
);

CREATE TABLE Maestros
(
MaestroID INT NOT NULL,
Especialidad VARCHAR(50),
PRIMARY KEY(MaestroID),
FOREIGN KEY(MaestroID) REFERENCES Usuarios(UsuarioID)
);


CREATE TABLE Alumnos
(
AlumnoID INT NOT NULL,
PRIMARY KEY(AlumnoID),
FOREIGN KEY(AlumnoID) REFERENCES Usuarios(UsuarioID)
);


CREATE TABLE Tutorias
(
TutoriaID INT NOT NULL AUTO_INCREMENT,
MaestroID INT,
AlumnoID INT,
PRIMARY KEY(TutoriaID),
FOREIGN KEY(MaestroID) REFERENCES Maestros(MaestroID),
FOREIGN KEY(AlumnoID) REFERENCES Alumnos(AlumnoID)
);
